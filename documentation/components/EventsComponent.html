   * @param {Object} event - Evento el cual va a mostrar la informacion mas detallada.
   */

  selectEvent(event: any) {
    this.selectedEvent = event;
  }

    /**
   * @method closeEvent
   * @description Cierra el evento seleccionado, estableciendo `selectedEvent` a `null`.

<div class="events-container">
    <h1>Eventos de Pokémon</h1>
    <div *ngFor="let event of eventsList" class="event-item">
      <img [src]="event.imageUrl" alt="{{ event.title }}" class="event-image">
      <div class="event-content">
        <h2>{{ event.title }}</h2>
        <p>{{ event.description }}</p>
        <small>{{ event.date }}</small>
        <div class="button-container">
        <button (click)="selectEvent(event)">Ver más</button>
        </div>
      </div>
    </div>
  
    <!-- Si seleccionamos algun evento,pues hacemos un ngIF que sirve para ponernos la condicion una vez que seleccionemos ese eventos y que nos muestre el titulo la imagen de ese evento seleccionado
